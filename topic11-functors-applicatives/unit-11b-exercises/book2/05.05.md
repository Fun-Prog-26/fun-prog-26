Q1 Solution:

reverseMaybe :: Maybe String -> Maybe String
reverseMaybe Nothing = Nothing
reverseMaybe (Just string) = Just (reverse string)

data Box a = Box a deriving Show

instance Functor Box where
   fmap func (Box val)  = Box (func val)
